def display_row(*objects,**options):
    class MultiObectRow(object):
        def _repr_html_(self):
            r =""
            format_str = ''.join(['{}:{};'.format(a,b) for a,b in options.items()])
            for obj in objects:
                if hasattr(obj,"_repr_html_"):
                    obj_str =obj._repr_html_()
                else: obj_str=str(obj)
                r+='<div class="col" style="{}">{}</div>'.format(format_str,obj_str)
            return '<div class="row" style="display: flex;">{}</div>'.format(r)
    return display(MultiObectRow())

def _ddf(df_data,caption = ""): #     pd.DF
    display (HTML("<h4>{2}</h4><p>({0}, {1})</p>".format(df_data.shape[0],df_data.shape[1],caption)))
    #.cm-s-ipython span.cm-variable
    if df_data.shape[1]>10:
        #display (HTML('<p class="code">{}</p>'.format(str(df_data.columns).replace('\n     ',''))))
        display (HTML('<p style="font-family: monospace;">{}</p>'.format(str(df_data.columns).replace('\n     ',''))))

    if df_data.shape[1]<4:
        display_row(df_data.head(5),df_data.sample(5).sort_index(),df_data.tail(5),margin ="1px 5px")
    elif(df_data.shape[1]<6):
        display_row(df_data.head(5),df_data.sample(5).sort_index(),margin ="1px 5px")
    elif(df_data.shape[1]<10):
        display(df_data.head(1).append(df_data.sample(5).sort_index()).append(df_data.tail(1)))
    else:
        display(df_data.head(1).append(df_data.sample(3).sort_index()).append(df_data.tail(1)).T)
